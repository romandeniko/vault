```sql
SELECT title, author, price, amount, // вывести общую стоимость книг
     price * amount AS total 
FROM book;
```

```sql
SELECT title, amount, amount * 1.65 AS pack // вывести название книги, ее количество и 
FROM book;                                  // стоимость упаковки при стоимости одного 
                                           // листа 1.65
```
