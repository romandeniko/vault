```sql
SELECT title, amount, price,                    // если количество книг меньше 4, то 
    IF(amount<4, price*0.5, price*0.7) AS sale  // скидка будет составлять 50% от цены, 
FROM book;                                      // в противном случае 30%.
```

```sql
SELECT title, amount, price,                            // с округлением
    ROUND(IF(amount<4, price*0.5, price*0.7),2) AS sale
FROM book;
```

```sql
SELECT title, amount, price,                         // Если количество книг меньше 4, 
    ROUND(                                           // то скидка составляет 50%, для 
     IF(amount < 4, price * 0.5,                     // остальных книг, количество 
         IF(amount < 11, price * 0.7, price * 0.9)), // которых меньше 11, скидка 30%, 
     2) AS sale,                                     // для всех оставшихся – 10%.
    IF(amount < 4, 'скидка 50%',                     // И еще укажем какая именно 
      IF(amount < 11, 'скидка 30%', 'скидка 10%')    // скидка на каждую книгу.
     ) AS Ваша_скидка
FROM book;
```

SELECT author, title,
